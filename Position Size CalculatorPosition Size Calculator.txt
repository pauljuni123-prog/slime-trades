class PositionSizeCalculator:
    def calculate(self, account_balance: Decimal, risk_percent: Decimal, 
                  stop_loss_pips: Decimal, pip_value: Decimal, 
                  symbol: str) -> dict:
        """
        Kelly Criterion-adjusted position sizing
        """
        # Base risk amount
        risk_amount = account_balance * (risk_percent / 100)
        
        # Kelly fraction (simplified)
        win_rate = self.get_user_win_rate(user_id, symbol, 30)
        avg_win = self.get_avg_win_r(user_id, symbol, 30)
        avg_loss = self.get_avg_loss_r(user_id, symbol, 30)
        
        kelly_fraction = win_rate - ((1 - win_rate) / (avg_win / avg_loss))
        kelly_fraction = max(0, min(kelly_fraction, 0.25))  # Cap at 25%
        
        # Adjusted risk
        adjusted_risk = risk_amount * kelly_fraction
        
        # Position size in lots
        risk_per_pip = adjusted_risk / stop_loss_pips
        position_size = risk_per_pip / pip_value
        
        return {
            "recommended_lots": round(position_size, 2),
            "risk_amount": adjusted_risk,
            "risk_percent_of_account": (adjusted_risk / account_balance) * 100,
            "kelly_fraction": kelly_fraction,
            "max_lots": self.get_max_lots(symbol, account_balance),
            "warning": position_size > self.get_max_lots(symbol, account_balance)
        }
