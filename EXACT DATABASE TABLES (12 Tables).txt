-- ============================================================
-- SLIME TRADES MVP — 12 Core Tables
-- PostgreSQL 16 + pgvector + pgcrypto
-- NO RLS (app-level filtering)
-- NO TimescaleDB (not needed at 100 users)
-- ============================================================

CREATE EXTENSION IF NOT EXISTS "uuid-ossp";
CREATE EXTENSION IF NOT EXISTS "pgcrypto";
CREATE EXTENSION IF NOT EXISTS "pgvector";

-- Enums
CREATE TYPE user_tier AS ENUM ('FREE', 'PRO', 'ELITE');
CREATE TYPE scan_type AS ENUM ('PRE', 'POST');
CREATE TYPE subscription_status AS ENUM ('TRIAL', 'ACTIVE', 'PAST_DUE', 'CANCELLED', 'EXPIRED');

-- 1. users
CREATE TABLE users (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    email VARCHAR(255) NOT NULL UNIQUE,
    password_hash TEXT NOT NULL,
    tier user_tier DEFAULT 'FREE',
    email_verified BOOLEAN DEFAULT FALSE,
    mfa_enabled BOOLEAN DEFAULT FALSE,
    mfa_secret_encrypted BYTEA,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW(),
    deleted_at TIMESTAMPTZ
);

-- 2. user_profiles
CREATE TABLE user_profiles (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    user_id UUID NOT NULL UNIQUE REFERENCES users(id) ON DELETE CASCADE,
    display_name VARCHAR(100),
    bio TEXT,
    trading_style VARCHAR(30),
    markets_traded TEXT[] DEFAULT '{}',
    risk_tolerance VARCHAR(30) DEFAULT 'MODERATE',
    psychological_triggers TEXT[] DEFAULT '{}',
    emergency_contact JSONB,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- 3. user_settings
CREATE TABLE user_settings (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    user_id UUID NOT NULL UNIQUE REFERENCES users(id) ON DELETE CASCADE,
    guardian_enabled BOOLEAN DEFAULT TRUE,
    daily_loss_limit DECIMAL(18,8) DEFAULT 500.00,
    max_daily_trades INT DEFAULT 10,
    mind_scan_required BOOLEAN DEFAULT FALSE,
    theme VARCHAR(10) DEFAULT 'DARK',
    language VARCHAR(5) DEFAULT 'en',
    notification_prefs JSONB DEFAULT '{"email": true, "push": false}',
    privacy_consents JSONB DEFAULT '{}',
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- 4. user_sessions
CREATE TABLE user_sessions (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    session_token_hash TEXT NOT NULL,
    ip_address INET,
    user_agent TEXT,
    expires_at TIMESTAMPTZ NOT NULL,
    revoked BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 5. subscription_plans
CREATE TABLE subscription_plans (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    plan_code VARCHAR(20) NOT NULL UNIQUE,
    plan_name VARCHAR(50) NOT NULL,
    price_monthly DECIMAL(10,2) NOT NULL,
    ai_messages_monthly INT NOT NULL,
    max_guardian_rules INT NOT NULL,
    max_journal_history_days INT,
    features JSONB DEFAULT '{}',
    is_active BOOLEAN DEFAULT TRUE
);

-- 6. subscriptions
CREATE TABLE subscriptions (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    user_id UUID NOT NULL UNIQUE REFERENCES users(id) ON DELETE CASCADE,
    plan_id UUID NOT NULL REFERENCES subscription_plans(id),
    status subscription_status DEFAULT 'TRIAL',
    stripe_customer_id VARCHAR(100),
    stripe_sub_id VARCHAR(100),
    current_period_end TIMESTAMPTZ,
    cancel_at_period_end BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- 7. trades
CREATE TABLE trades (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    symbol VARCHAR(20) NOT NULL,
    direction VARCHAR(4) NOT NULL CHECK (direction IN ('BUY', 'SELL')),
    entry_price DECIMAL(18,8) NOT NULL,
    exit_price DECIMAL(18,8),
    volume DECIMAL(18,8) NOT NULL,
    realized_pnl DECIMAL(18,8) DEFAULT 0,
    commission DECIMAL(18,8) DEFAULT 0,
    swap DECIMAL(18,8) DEFAULT 0,
    net_pnl DECIMAL(18,8) GENERATED ALWAYS AS (COALESCE(realized_pnl,0) - COALESCE(commission,0) - COALESCE(swap,0)) STORED,
    strategy_tag VARCHAR(50),
    emotion_tag VARCHAR(50),
    notes TEXT,
    opened_at TIMESTAMPTZ NOT NULL,
    closed_at TIMESTAMPTZ,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 8. mind_scans
CREATE TABLE mind_scans (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    scan_type scan_type NOT NULL DEFAULT 'PRE',
    stress_level INT NOT NULL CHECK (stress_level BETWEEN 1 AND 10),
    confidence_score INT NOT NULL CHECK (confidence_score BETWEEN 1 AND 10),
    fomo_level INT NOT NULL CHECK (fomo_level BETWEEN 1 AND 5),
    revenge_urge INT NOT NULL CHECK (revenge_urge BETWEEN 1 AND 5),
    fatigue_level INT NOT NULL CHECK (fatigue_level BETWEEN 1 AND 5),
    readiness_score INT GENERATED ALWAYS AS (
        GREATEST(0, LEAST(100,
            100
            - (stress_level * 10)
            - (fomo_level * 8)
            - (revenge_urge * 15)
            - (fatigue_level * 5)
            + (confidence_score * 2)
        ))
    ) STORED,
    notes TEXT,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 9. journal_entries
CREATE TABLE journal_entries (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    trade_id UUID REFERENCES trades(id) ON DELETE SET NULL,
    entry_type VARCHAR(20) DEFAULT 'DAILY',
    narrative TEXT,
    what_i_saw TEXT,
    what_i_felt TEXT,
    what_i_did TEXT,
    what_next TEXT,
    mood_rating INT CHECK (mood_rating BETWEEN 1 AND 10),
    tags TEXT[] DEFAULT '{}',
    ai_summary TEXT,
    ai_sentiment VARCHAR(20),
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- 10. sanctuary_sessions
CREATE TABLE sanctuary_sessions (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    technique VARCHAR(30) NOT NULL DEFAULT 'BOX_BREATHING',
    duration_seconds INT NOT NULL DEFAULT 180,
    completed BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 11. ai_conversations
CREATE TABLE ai_conversations (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    session_id VARCHAR(50) NOT NULL,
    title VARCHAR(100),
    is_active BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW(),
    UNIQUE(user_id, session_id)
);

-- 12. ai_messages
CREATE TABLE ai_messages (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    conversation_id UUID NOT NULL REFERENCES ai_conversations(id) ON DELETE CASCADE,
    user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    role VARCHAR(10) NOT NULL CHECK (role IN ('USER', 'AI')),
    content TEXT NOT NULL,
    context_vector VECTOR(1536),
    tokens_used INT,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- ============================================================
-- INDEXES
-- ============================================================

CREATE INDEX idx_users_email ON users(email);
CREATE INDEX idx_users_tier ON users(tier);

CREATE INDEX idx_trades_user_date ON trades(user_id, opened_at DESC);
CREATE INDEX idx_trades_user_symbol ON trades(user_id, symbol);

CREATE INDEX idx_mind_scans_user ON mind_scans(user_id, created_at DESC);
CREATE INDEX idx_mind_scans_readiness ON mind_scans(user_id, readiness_score);

CREATE INDEX idx_journal_user ON journal_entries(user_id, created_at DESC);
CREATE INDEX idx_journal_fts ON journal_entries USING gin(to_tsvector('english', COALESCE(narrative, '')));

CREATE INDEX idx_sanctuary_user ON sanctuary_sessions(user_id, created_at DESC);

CREATE INDEX idx_ai_conv_user ON ai_conversations(user_id, updated_at DESC);
CREATE INDEX idx_ai_msg_conv ON ai_messages(conversation_id, created_at DESC);
CREATE INDEX idx_ai_msg_vector ON ai_messages USING ivfflat (context_vector vector_cosine_ops) WITH (lists = 50);

CREATE INDEX idx_subscriptions_user ON subscriptions(user_id);
CREATE INDEX idx_subscriptions_stripe ON subscriptions(stripe_sub_id);

-- ============================================================
-- TRIGGERS
-- ============================================================

CREATE OR REPLACE FUNCTION update_updated_at()
RETURNS TRIGGER AS $$
BEGIN
    NEW.updated_at = NOW();
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER tr_users_updated BEFORE UPDATE ON users FOR EACH ROW EXECUTE FUNCTION update_updated_at();
CREATE TRIGGER tr_profiles_updated BEFORE UPDATE ON user_profiles FOR EACH ROW EXECUTE FUNCTION update_updated_at();
CREATE TRIGGER tr_settings_updated BEFORE UPDATE ON user_settings FOR EACH ROW EXECUTE FUNCTION update_updated_at();
CREATE TRIGGER tr_subscriptions_updated BEFORE UPDATE ON subscriptions FOR EACH ROW EXECUTE FUNCTION update_updated_at();
CREATE TRIGGER tr_journal_updated BEFORE UPDATE ON journal_entries FOR EACH ROW EXECUTE FUNCTION update_updated_at();
CREATE TRIGGER tr_ai_conv_updated BEFORE UPDATE ON ai_conversations FOR EACH ROW EXECUTE FUNCTION update_updated_at();
